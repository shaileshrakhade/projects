create database stock_prediction;

use stock_prediction;

create table user
(
id int primary key auto_increment,
name varchar(20),
pass varchar(20),
type varchar(10) default 'user'
);


create table company_name
(
id int primary key auto_increment,
cname varchar(30),
oldrate int,
newrte int,
persent int
);

create table stock_rate
(
id int primary key auto_increment,
cid int not null,
FOREIGN KEY(cid)REFERENCES company_name(id),
date varchar(30),
rate int,
news_rate int
);
